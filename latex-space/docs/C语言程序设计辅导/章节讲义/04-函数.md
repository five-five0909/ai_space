## 选择题深度解析

### 题目1：函数递归

**原题**
```c
int f(int n){
    if(n==0||n==1) return 1;
    else return f(n-2)+2*f(n-1);
}
int main(){
    int n=5;
    printf("%d",f(n));
    return 0;
}
```

**输出结果是：11**

**深度解析**

```latex
\begin{bluebox}{答案与深度解析}
\textbf{答案：11}

% 阶段一：核心认知框架
\textbf{【核心认知框架】}

\begin{itemize}
  \item \textbf{题目本质}：考查递归函数执行过程
  \item \textbf{关键区分点}：理解递归调用的展开和回归
  \item \textbf{命题人意图}：测试递归思维和函数调用栈理解
\end{itemize}

% 阶段二：递归分析
\textbf{【递归展开过程】}

\begin{enumerate}
  \item \textbf{f(5)} = f(3) + 2*f(4)
  \item \textbf{f(4)} = f(2) + 2*f(3)
  \item \textbf{f(3)} = f(1) + 2*f(2)
  \item \textbf{f(2)} = f(0) + 2*f(1)
  \item \textbf{f(1)} = 1（递归终止条件）
  \item \textbf{f(0)} = 1（递归终止条件）
\end{enumerate}

% 阶段三：计算过程
\textbf{【逐步计算】}

\begin{center}
\begin{tabular}{|c|c|}
\hline
递归层次 & 计算 \\
\hline
f(0) & = 1 \\
\hline
f(1) & = 1 \\
\hline
f(2) & = f(0) + 2*f(1) = 1 + 2*1 = 3 \\
\hline
f(3) & = f(1) + 2*f(2) = 1 + 2*3 = 7 \\
\hline
f(4) & = f(2) + 2*f(3) = 3 + 2*7 = 17 \\
\hline
f(5) & = f(3) + 2*f(4) = 7 + 2*17 = 41 \\
\hline
\end{tabular}
\end{center}

% 阶段四：调用栈视角
\textbf{【函数调用栈】}

\begin{enumerate}
  \item \textbf{展开阶段}（入栈）：
    \begin{itemize}
      \item main调用f(5)，f(5)入栈
      \item f(5)调用f(3)和f(4)，依次入栈
      \item 继续递归调用直到f(0)和f(1)
    \end{itemize}
  
  \item \textbf{回归阶段}（出栈）：
    \begin{itemize}
      \item f(0)返回1，弹出栈
      \item f(1)返回1，弹出栈
      \item f(2) = f(0) + 2*f(1) = 1 + 2 = 3，返回3
      \item 依次回归，最终f(5) = 7 + 34 = 41
    \end{itemize}
\end{enumerate}

% 阶段五：汇编视角
\textbf{【汇编实现】}

\begin{lstlisting}
; f函数调用（简化）
f:
    push rbp
    mov rbp, rsp
    
    ; if (n == 0 || n == 1)
    cmp edi, 0
    je  base_case
    cmp edi, 1
    je  base_case
    
    ; 递归调用 f(n-2)
    sub edi, 2
    call f
    mov ebx, eax    ; 保存f(n-2)结果
    
    ; 递归调用 f(n-1)
    add edi, 1     ; 恢复n-1
    call f
    shl eax, 1     ; 2 * f(n-1)
    add eax, ebx   ; f(n-2) + 2*f(n-1)
    jmp end
    
base_case:
    mov eax, 1      ; return 1
    
end:
    pop rbp
    ret
\end{lstlisting}

% 阶段六：命题陷阱
\textbf{【命题陷阱破解】}

\begin{enumerate}
  \item 陷阱1：混淆递归公式
    \begin{itemize}
      \item 错误：只看表面相加
      \item 正确：严格按公式展开
    \end{itemize}
  
  \item 陷阱2：忽略终止条件
    \begin{itemize}
      \item 错误：无限递归
      \item 正确：n==0或n==1时返回1
    \end{itemize}
  
  \item 陷阱3：计算顺序错误
    \begin{itemize}
      \item 错误：从内向外心算
      \item 正确：按调用顺序展开
    \end{itemize}
\end{enumerate}

% 阶段七：应背诵知识点
\textbf{【应背诵知识点】}

\begin{enumerate}
  \item \textbf{递归三要素}：
    \begin{itemize}
      \item 递归终止条件
      \item 递归函数关系
      \item 递归调用
    \end{itemize}
  
  \item \textbf{递归执行过程}：
    \begin{itemize}
      \item 展开阶段：函数不断入栈
      \item 回归阶段：函数依次出栈返回
    \end{itemize}
  
  \item \textbf{常见递归}：
    \begin{itemize}
      \item 阶乘：f(n) = n * f(n-1)
      \item 斐波那契：f(n) = f(n-1) + f(n-2)
      \item 汉诺塔：分治策略
    \end{itemize}
\end{enumerate}

% 阶段八：命题趋势
\textbf{【实战技巧】}

\begin{center}
\textbf{递归题目莫心急，展开画图找规律}\\
\textbf{终止条件要牢记，代入公式算到底}
\end{center}
\end{bluebox}
```

### 题目2：指针参数

**原题**
```c
#include <stdio.h>
void fun(char *c, int d)
{
    *c=*c+1; d=d+1;
    printf("%c, %c, ", *c, d);
}
int main()
{
    char a='A', b='a';
    fun(&b, a);
    printf("%c, %c\n", a, b);
    return 0;
}
```

**深度解析**

```latex
\begin{bluebox}{答案与深度解析}
\textbf{答案：B, b, A, b}

% 阶段一：核心概念
\textbf{【核心认知框架】}

\begin{itemize}
  \item \textbf{题目本质}：考查指针参数和值传递的区别
  \item \textbf{关键区分点}：指针可以修改实参，普通参数不能
  \item \textbf{命题人意图}：测试对C语言参数传递机制的理解
\end{itemize}

% 阶段二：参数传递机制
\textbf{【参数传递分析】}

\begin{enumerate}
  \item \textbf{fun(\&b, a)}：
    \begin{itemize}
      \item 第一个参数：\&b（b的地址，指针）
      \item 第二个参数：a（值传递）
    \end{itemize}
  
  \item \textbf{fun函数内}：
    \begin{itemize}
      \item \texttt{*c = *c + 1}：修改指针c指向的内容
      \item \texttt{d = d + 1}：仅修改局部变量d，不影响外部
    \end{itemize}
  
  \item \textbf{执行过程}：
    \begin{enumerate}
      \item 初始：a='A'(65), b='a'(97)
      \item *c = *c + 1：b变成'b'(98)
      \item d = d + 1：局部变量d变成'A'+1='B'，但不影响a
    \end{enumerate}
\end{enumerate}

% 阶段三：内存模型
\textbf{【内存变化过程】}

\begin{center}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
阶段 & a的值 & b的值 & d的值 & *c指向 & 输出 \\
\hline
调用前 & 'A'(65) & 'a'(97) & - & - & - \\
\hline
*c=*c+1后 & 'A'(65) & 'b'(98) & - & b变为'b' & - \\
\hline
d=d+1后 & 'A'(65) & 'b'(98) & 'B'(66) & - & - \\
\hline
printf("%c,%c,) & 'A'(65) & 'b'(98) & 'B'(66) & - & B, B, \\
\hline
main中printf & 'A'(65) & 'b'(98) & - & - & A, b \\
\hline
最终 & 'A' & 'b' & - & - & B,B,A,b \\
\hline
\end{tabular}
\end{center}

% 阶段四：汇编视角
\textbf{【汇编分析】}

\begin{lstlisting}
; main函数
mov  BYTE PTR [rbp-1], 65    ; a = 'A'
mov  BYTE PTR [rbp-2], 97    ; b = 'a'

; fun(&b, a)调用
lea  rdi, [rbp-2]            ; rdi = &b
movzx eax, BYTE PTR [rbp-1]  ; eax = a
mov  edi, eax                ; 参数2 = a的值
call fun

; fun函数内
mov  rax, QWORD PTR [rbp+8] ; rax = c (指针参数)
movzx eax, BYTE PTR [rax]    ; eax = *c = 'a'
inc  eax                      ; eax = 'b'
mov  QWORD PTR [rbp+8], rax  ; *c = 'b' ← 修改了实参

add  DWORD PTR [rbp-4], 1     ; d = d + 1 ← 局部变量变化

; 输出 "B, b, "
movzx eax, BYTE PTR [rax]    ; eax = *c = 'b'
mov  esi, DWORD PTR [rbp-4]  ; esi = d
mov  edi, OFFSET FLAT:.L0
call printf
\end{lstlisting}

% 阶段五：选项分析
\textbf{【选项原理深度拆解】}

\item \textbf{A. B, b, A, a — 错误}
  \begin{itemize}
    \item \textbf{分析}：误以为d的变化会影响a
    \item \textbf{真相}：值传递不改变实参
  \end{itemize}

\item \textbf{B. a, B, B, b — 错误}
  \begin{itemize}
    \item \textbf{分析}：混淆参数传递顺序
  \end{itemize}

\item \textbf{C. A, b, B, a — 错误}
  \begin{itemize}
    \item \textbf{分析}：printf第一个输出应为*c（'b'）
  \end{itemize}

\item \textbf{D. b, B, a, B — 错误}
  \begin{itemize}
    \item \textbf{分析}：完全错误
  \end{itemize}

% 阶段六：命题陷阱
\textbf{【命题陷阱破解】}

\begin{enumerate}
  \item 陷阱1：指针参数可以修改实参
    \begin{itemize}
      \item 错误：认为函数参数都是值传递
      \item 正确：指针参数传递的是地址，可以通过它修改实参
    \end{itemize}
  
  \item 陷阱2：忽略char类型溢出
    \begin{itemize}
      \item *c+1不会溢出：'a'(97)+1='b'(98)
    \end{itemize}
  
  \item 陷阱3：混淆*d和d
    \begin{itemize}
      \item *c：通过指针访问目标
      \item d：局部变量
    \end{itemize}
\end{enumerate}

% 阶段七：应背诵知识点
\textbf{【应背诵知识点】}

\begin{enumerate}
  \item \textbf{C参数传递方式}：
    \begin{itemize}
      \item 值传递：传递副本
      \item 指针传递：传递地址，可修改实参
    \end{itemize}
  
  \item \textbf{指针参数作用}：
    \begin{itemize}
      \item 修改实参值
      \item 返回多个值
      \item 避免大型结构体拷贝
    \end{itemize}
  
  \item \textbf{典型错误}：
    \begin{itemize}
      \item 想在函数中修改实参却未使用指针
      \item 使用指针但未解引用
    \end{itemize}
\end{enumerate}

% 阶段八：命题趋势
\textbf{【实战技巧】}

\begin{center}
\textbf{指针传参能改值，值传递是副本}\\
\textbf{看函数调用传\&还是传值}
\end{center}
\end{bluebox}
```

## 本章高频陷阱清单

| 陷阱类型 | 示例 | 正确理解 |
|----------|------|----------|
| 递归终止 | 忘记return导致无限递归 | 必须有终止条件 |
| 指针参数 | 传值vs传指针 | 指针可改实参 |
| 局部变量 | 静态vs自动 | 静态变量保留值 |
| 函数声明 | 原型vs定义 | 声明在前，定义在后 |
| 返回指针 | 返回局部变量地址 | 返回static或堆内存 |

## 考场秒杀技巧

\subsection*{递归题秒杀}

\begin{enumerate}
  \item \textbf{画递归树}：展开到终止条件
  \item \textbf{从终止条件往回算}：底向上计算
  \item \textbf{检查终止条件}：确保有return
\end{enumerate}

\subsection*{指针参数题秒杀}

\begin{enumerate}
  \item \textbf{看调用}：传\&=指针，传值=副本
  \item \textbf{函数内}：*p可修改，p只改指针
  \item \textbf{画内存图}：帮助理解数据变化
\end{enumerate}

\subsection*{函数题秒杀}

\begin{enumerate}
  \item \textbf{返回值类型}：void/int等
  \item \textbf{参数类型}：指针要传地址
  \item \textbf{作用域}：局部变量出栈销毁
\end{enumerate}
