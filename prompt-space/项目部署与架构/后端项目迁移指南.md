# MIGRATION_GUIDE.md

## 项目迁移指南：Servlet/JSP → Spring Boot + MyBatis-Plus

---

## 一、迁移概述

### 1.1 项目信息

| 项目 | 原技术栈 | 目标技术栈 |
|------|---------|-----------|
| 新闻管理系统后端 | Servlet 3.0 + JSP + JDBC | Java 17 + Spring Boot 3.x + MyBatis-Plus 3.5 + MySQL 8.0 |

### 1.2 迁移范围

- **后端重构**: Servlet → Spring Boot REST API
- **数据访问**: JDBC → MyBatis-Plus
- **响应格式**: JSP 视图 → JSON API
- **保持不变**: 数据库结构、前端 JSP 页面

### 3.2 统一响应格式

```java
// Result.java
@Data
public class Result<T> {
    private Integer code;      // 200=成功, 500=失败
    private String msg;
    private T data;

    public static <T> Result<T> success() {
        return success(null);
    }

    public static <T> Result<T> success(T data) {
        Result<T> r = new Result<>();
        r.setCode(200);
        r.setMsg("success");
        r.setData(data);
        return r;
    }

    public static <T> Result<T> error(String msg) {
        Result<T> r = new Result<>();
        r.setCode(500);
        r.setMsg(msg);
        return r;
    }
}
```

## 七、Mapper 层实现

### 7.1 NewsMapper.java

```java
package com.artisan.mapper;

import com.artisan.entity.News;
import com.baomidou.mybatisplus.core.mapper.BaseMapper;
import org.apache.ibatis.annotations.Mapper;
import org.apache.ibatis.annotations.Param;
import org.apache.ibatis.annotations.Select;

import java.util.List;

@Mapper
public interface NewsMapper extends BaseMapper<News> {

    /**
     * 查询新闻列表（带分类名称）
     */
    @Select("""
        <script>
        SELECT t1.*, t2.typeName
        FROM t_news t1
        LEFT JOIN t_newsType t2 ON t1.typeId = t2.newsTypeId
        <where>
            <if test="typeId != null and typeId != -1">
                AND t1.typeId = #{typeId}
            </if>
            <if test="title != null and title != ''">
                AND t1.title LIKE CONCAT('%', #{title}, '%')
            </if>
            <if test="s_bPublishDate != null">
                AND DATE(t1.publishDate) >= #{s_bPublishDate}
            </if>
            <if test="s_aPublishDate != null">
                AND DATE(t1.publishDate) &lt;= #{s_aPublishDate}
            </if>
        </where>
        ORDER BY t1.publishDate DESC
        LIMIT #{offset}, #{pageSize}
        </script>
        """)
    List<News> selectNewsListWithType(@Param("typeId") Integer typeId,
                                       @Param("title") String title,
                                       @Param("s_bPublishDate") String s_bPublishDate,
                                       @Param("s_aPublishDate") String s_aPublishDate,
                                       @Param("offset") int offset,
                                       @Param("pageSize") int pageSize);

    /**
     * 统计新闻数量
     */
    @Select("""
        <script>
        SELECT COUNT(*)
        FROM t_news
        <where>
            <if test="typeId != null and typeId != -1">
                AND typeId = #{typeId}
            </if>
            <if test="title != null and title != ''">
                AND title LIKE CONCAT('%', #{title}, '%')
            </if>
            <if test="s_bPublishDate != null">
                AND DATE(publishDate) >= #{s_bPublishDate}
            </if>
            <if test="s_aPublishDate != null">
                AND DATE(publishDate) &lt;= #{s_aPublishDate}
            </if>
        </where>
        </script>
        """)
    int countNews(@Param("typeId") Integer typeId,
                  @Param("title") String title,
                  @Param("s_bPublishDate") String s_bPublishDate,
                  @Param("s_aPublishDate") String s_aPublishDate);

    /**
     * 查询上一条/下一条新闻
     */
    @Select("""
        (SELECT * FROM t_news WHERE newsId &lt; #{newsId} ORDER BY newsId DESC LIMIT 1)
        UNION ALL
        (SELECT * FROM t_news WHERE newsId &gt; #{newsId} ORDER BY newsId ASC LIMIT 1)
        """)
    List<News> getUpAndDownPageId(@Param("newsId") Integer newsId);

    /**
     * 根据ID查询（带分类名称）
     */
    @Select("""
        SELECT t1.*, t2.typeName
        FROM t_news t1
        LEFT JOIN t_newsType t2 ON t1.typeId = t2.newsTypeId
        WHERE t1.newsId = #{newsId}
        """)
    News selectByIdWithType(@Param("newsId") Integer newsId);

    /**
     * 增加点击量
     */
    @Select("UPDATE t_news SET click = click + 1 WHERE newsId = #{newsId}")
    int increaseClick(@Param("newsId") Integer newsId);

    /**
     * 检查分类下是否存在新闻
     */
    @Select("SELECT COUNT(*) FROM t_news WHERE typeId = #{typeId}")
    int existNewsWithNewsTypeId(@Param("typeId") Integer typeId);
}
```

### 8.2 NewsServiceImpl.java

```java
package com.artisan.service.impl;

import com.artisan.entity.News;
import com.artisan.mapper.NewsMapper;
import com.artisan.service.NewsService;
import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.*;

@Service
public class NewsServiceImpl extends ServiceImpl<NewsMapper, News> implements NewsService {

    @Autowired
    private NewsMapper newsMapper;

    @Override
    public List<News> getNewsList(Integer typeId, String title, String s_bPublishDate,
                                   String s_aPublishDate, int page, int pageSize) {
        int offset = (page - 1) * pageSize;
        return newsMapper.selectNewsListWithType(typeId, title, s_bPublishDate,
                                                   s_aPublishDate, offset, pageSize);
    }

    @Override
    public int getNewsCount(Integer typeId, String title, String s_bPublishDate,
                            String s_aPublishDate) {
        return newsMapper.countNews(typeId, title, s_bPublishDate, s_aPublishDate);
    }

    @Override
    public News getNewsById(Integer newsId) {
        return newsMapper.selectByIdWithType(newsId);
    }

    @Override
    public void increaseClick(Integer newsId) {
        newsMapper.increaseClick(newsId);
    }

    @Override
    public Map<String, News> getUpAndDownPage(Integer newsId) {
        List<News> list = newsMapper.getUpAndDownPageId(newsId);
        Map<String, News> result = new HashMap<>();
        if (list.size() >= 2) {
            result.put("upNews", list.get(0));
            result.put("downNews", list.get(1));
        }
        return result;
    }

    @Override
    public boolean saveNews(News news) {
        if (news.getPublishDate() == null) {
            news.setPublishDate(new Date());
        }
        if (news.getClick() == null) {
            news.setClick(0);
        }
        return this.saveOrUpdate(news);
    }

    @Override
    public boolean deleteNews(Integer newsId) {
        return this.removeById(newsId);
    }

    @Override
    public List<News> getImageNews() {
        LambdaQueryWrapper<News> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(News::getIsImage, 1)
               .orderByDesc(News::getPublishDate)
               .last("LIMIT 5");
        return this.list(wrapper);
    }

    @Override
    public News getHeadNews() {
        LambdaQueryWrapper<News> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(News::getIsHead, 1)
               .orderByDesc(News::getPublishDate)
               .last("LIMIT 1");
        return this.getOne(wrapper);
    }

    @Override
    public List<News> getHotNews(int limit) {
        LambdaQueryWrapper<News> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(News::getIsHot, 1)
               .orderByDesc(News::getPublishDate)
               .last("LIMIT " + limit);
        return this.list(wrapper);
    }

    @Override
    public List<News> getNewsByTypeId(Integer typeId, int limit) {
        LambdaQueryWrapper<News> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(News::getTypeId, typeId)
               .orderByDesc(News::getPublishDate)
               .last("LIMIT " + limit);
        return this.list(wrapper);
    }
}
```

```java
package com.artisan.controller;

import com.artisan.entity.Comment;
import com.artisan.entity.News;
import com.artisan.service.CommentService;
import com.artisan.service.NewsService;
import com.artisan.common.Result;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

@RestController
@RequestMapping("/news")
@CrossOrigin
public class NewsController {

    @Autowired
    private NewsService newsService;

    @Autowired
    private CommentService commentService;

    /**
     * 前台新闻列表
     */
    @GetMapping("/list")
    public Result<Map<String, Object>> list(
            @RequestParam(required = false) Integer typeId,
            @RequestParam(defaultValue = "1") Integer page) {
        int pageSize = 10;
        List<News> newsList = newsService.getNewsList(typeId, null, null, null, page, pageSize);
        int total = newsService.getNewsCount(typeId, null, null, null);

        Map<String, Object> data = new HashMap<>();
        data.put("newsList", newsList);
        data.put("total", total);
        data.put("page", page);
        data.put("pageSize", pageSize);
        return Result.success(data);
    }

    /**
     * 前台新闻详情
     */
    @GetMapping("/show/{newsId}")
    public Result<Map<String, Object>> show(@PathVariable Integer newsId) {
        Map<String, Object> data = new HashMap<>();

        News news = newsService.getNewsById(newsId);
        newsService.increaseClick(newsId);

        List<Comment> commentList = commentService.getCommentsByNewsId(newsId);
        Map<String, News> upDownPage = newsService.getUpAndDownPage(newsId);

        data.put("news", news);
        data.put("commentList", commentList);
        data.put("upNews", upDownPage.get("upNews"));
```

```java
package com.artisan.controller;

import com.artisan.common.Result;
```

```java
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;
import org.springframework.web.multipart.MultipartHttpServletRequest;

import javax.servlet.http.HttpServletRequest;
import java.io.File;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;

@RestController
@RequestMapping("/image")
@CrossOrigin
public class ImageUploadController {

    @Value("${app.image-path}")
    private String imagePath;

    /**
     * CKEditor 图片上传
     */
    @PostMapping("/upload")
    public Map<String, Object> upload(HttpServletRequest request) {
        Map<String, Object> result = new HashMap<>();

        try {
            MultipartHttpServletRequest multipartRequest = (MultipartHttpServletRequest) request;
            MultipartFile file = multipartRequest.getFile("upload");

            if (file != null && !file.isEmpty()) {
                String originalFilename = file.getOriginalFilename();
                String ext = originalFilename.substring(originalFilename.lastIndexOf(".") + 1);
                String fileName = new SimpleDateFormat("yyyyMMddHHmmss").format(new Date()) + "." + ext;

                File targetFile = new File(imagePath + fileName);
                file.transferTo(targetFile);

                result.put("uploaded", true);
                result.put("url", "userImage/" + fileName);
            } else {
                result.put("uploaded", false);
                result.put("error", "上传文件为空");
            }
        } catch (Exception e) {
            result.put("uploaded", false);
            result.put("error", "上传失败: " + e.getMessage());
        }

        return result;
    }
}
```

### 11.1 PageUtil.java

```java
package com.artisan.util;

public class PageUtil {

    /**
     * 生成分页代码
     * @param total 总记录数
     * @param page 当前页
     * @param pageSize 每页大小
     * @param url 基础URL
     * @return 分页HTML
     */
    public static String getPagination(int total, int page, int pageSize, String url) {
        int totalPage = (total + pageSize - 1) / pageSize;
        StringBuilder sb = new StringBuilder();

        if (page > 1) {
            sb.append("<a href='").append(url).append("?page=").append(page - 1)
              .append("'>上一页</a>&nbsp;");
        }

        for (int i = 1; i <= totalPage; i++) {
            if (i == page) {
                sb.append("<span class='current'>").append(i).append("</span>&nbsp;");
            } else {
                sb.append("<a href='").append(url).append("?page=").append(i)
                  .append("'>").append(i).append("</a>&nbsp;");
            }
        }

        if (page < totalPage) {
            sb.append("<a href='").append(url).append("?page=").append(page + 1)
              .append("'>下一页</a>");
        }

        return sb.toString();
    }

    /**
     * 生成分页数据
     */
    public static Map<String, Object> getPageData(int total, int page, int pageSize) {
        Map<String, Object> data = new HashMap<>();
        int totalPage = (total + pageSize - 1) / pageSize;
        data.put("total", total);
        data.put("page", page);
        data.put("pageSize", pageSize);
        data.put("totalPage", totalPage);
        return data;
    }
}
```

```java
package com.artisan.util;

import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;

public class DateUtil {

    private static final String FORMAT = "yyyy-MM-dd HH:mm:ss";

    public static String formatDate(Date date) {
        if (date == null) return "";
        return new SimpleDateFormat(FORMAT).format(date);
    }

    public static Date parseDate(String dateStr) {
        if (dateStr == null || dateStr.trim().isEmpty()) return null;
        try {
            return new SimpleDateFormat(FORMAT).parse(dateStr);
        } catch (ParseException e) {
            return null;
        }
    }

    public static String getCurrentDateStr() {
        return new SimpleDateFormat("yyyyMMddHHmmss").format(new Date());
    }
}
```
